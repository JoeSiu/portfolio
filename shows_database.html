Shows/movies database

A database web app using AppSheet, Google AppScript and Google Sheet 

Previous Next

Details

This web app is a personal side project which stores the TV shows/ movies which titles that I have watched or wants to watch later. This allows me to easily review and update the records. AppSheet is used as the front end interface of the web app, while Google Sheet is used as the data storage, and Google AppScript for timed custom events and functions.

App features

  • Add/ remove records
  • Shows type (season 1, season 2, live-action, etc)
  • Shows status (watched, finished, unreleased, etc) 
  • Watch order
  • Current episode (for unfinished shows)
  • Rating (1-10, for finished shows)
  • Remarks
  • Featuring (whether it will be featured on the homepage of the app)
  • Statistics
  • Auto get wiki link based on the show title

function submitData() {
    var ss        = SpreadsheetApp.getActiveSpreadsheet();

    var datasheet = ss.getSheetByName("Overview"); //Data Sheet

  function setBorder() {
    //script from: https://stackoverflow.com/questions/55943889/adding-borders-based-on-grouped-rows-and-conditional-formatting-in-google-sheets
    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var datasheet = ss.getSheetByName("Overview"); //Data Sheet
    
    //clear border
    datasheet.getRange(1, 1, datasheet.getMaxRows(), datasheet.getMaxColumns()).setBorder(false, false, false, false, false, false);
    
    //get IDs, getRange(row, column...
    var values = datasheet.getRange(2, 1, datasheet.getLastRow() - 1, 1).getValues();
  
    //Check if current ID is same as next ID, if so, add it to rangeList
    var rangeList = values.reduce(function(ar, e, i) {
        if (i > 0 && values[i - 1][0] != e[0] && e[0] != "") {
          //range, from Ai+1:Ki+1
          ar.push("A" + (i + 1) + ":M" + (i + 1));  //if added more column, change the ":x"
        }
        return ar;
      }, [])
    
    rangeList.push(datasheet.getRange(datasheet.getLastRow(), 1, 1, datasheet.getLastColumn()).getA1Notation());
    
    //set borders, setBorder(top, left, bottom, right, vertical, horizontal, color, style)
    var color = 'dimgrey';
    var style = 'SOLID';
    datasheet.getRangeList(rangeList).setBorder(null, null, true, null, false, false, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('A1:K1').setBorder(true, true, true, true, true, null, color, SpreadsheetApp.BorderStyle.style);  //~if added more column, change the range value~
    datasheet.getRange('A2:A').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('D2:D').setBorder(null, null, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('G2:G').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('H2:H').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);  
    datasheet.getRange('J2:J').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('K2:K').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('L2:L').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);
    datasheet.getRange('M2:M').setBorder(null, true, null, true, null, null, color, SpreadsheetApp.BorderStyle.style);

  }

  function setID() {
    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var datasheet = ss.getSheetByName("Overview");
    var cells = datasheet.getRange(2, 1, datasheet.getLastRow()-1, 1).getValues();
    
    sortRange();
    SpreadsheetApp.flush();
    
     for (var i = 1; i < cells.length-1; i++) { 
       var previous = cells[i-1][0];
       var current = cells[i][0];
       if (previous != '' && current != '') {
         var interval = current - previous;
         
         if (interval > 0) {
           interval--;
         }
         
         while(cells[i][0] == current && i < cells.length-1) {
           cells[i][0] = (current - interval);
           i++;
         } 
         i--;
       }
     }
    
     var previous = cells[cells.length-2][0];
     var current = cells[cells.length-1][0];
     var interval = current - previous;
       
     if (interval > 0) {
       interval--;
     }
    cells[cells.length-1][0] = (current - interval);
    
    datasheet.getRange(2, 1,  cells.length, 1).setValues(cells);
  }

  function searchWiki() {
    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var datasheet = ss.getSheetByName("Overview");
    var cells = datasheet.getRange(2, 13, datasheet.getLastRow()-1).getValues(); //get row 2, column 13's data
   
     for (var i = 0; i<cells.length; i++) { 
       //if cell is empty or is error
       if (cells[i] == "" || cells[i] == "#ERROR!") {
         datasheet.getRange(i+2,13).setFormula("=CONCATENATE(\"https://zh.wikipedia.org\",IFNA(index(importxml(\"https://zh.wikipedia.org/w/index.php?search=\"" + "&$E" + (i+2) + ",\"//div[@id='mw-content-text']/div[@class='searchresults mw-searchresults-has-iw']/ul[@class='mw-search-results']//a/@href\"),1,1)," + "CONCATENATE(\"/w/index.php?search=\"," + "$E" + (i+2) + ")))"); 
         //SpreadsheetApp.flush();
         Logger.log('Row ' + (i+2) + ' have been added the formula');  
       }   
       //if the cell contains a formula
       else if (datasheet.getRange(2+i,13).getFormula()) {
         //get the value from the formula and set the value to the cell
         var val = datasheet.getRange(i+2,13).getValue();
         datasheet.getRange(i+2,13).setValue(val);
         //SpreadsheetApp.flush();
         Logger.log('Row ' + (i+2) + ' have been set');  
       }
    }
  }     
  
  //Remove All Rows in Overview which the title is empty
  function removeRowsWithNoTitle() {
    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var datasheet = ss.getSheetByName("Overview");
    var rows = datasheet.getDataRange();
    var numRows = rows.getNumRows();
    var values = rows.getValues();
  
    var rowsDeleted = 0;
    for (var i = 0; i <= numRows - 1; i++) {
      var row = values[i];
      if (row[4] == '') { //title is at row 5, array start at 0, so 4
        datasheet.deleteRow((parseInt(i)+1) - rowsDeleted);
        rowsDeleted++;
      }
    }
  }
  
  //Remove All Empty Rows in the Entire Workbook
  function removeEmptyRows() {
    var ss = SpreadsheetApp.getActive();
    var allsheets = ss.getSheets();
    for (var s in allsheets){
      var sheet=allsheets[s]
      var maxRows = sheet.getMaxRows(); 
      var lastRow = sheet.getLastRow();
      if (maxRows-lastRow != 0){
        sheet.deleteRows(lastRow+1, maxRows-lastRow);
      }
    }
  }
  
  //Remove All Empty Columns in the Entire Workbook
  function removeEmptyColumns() {
    var ss = SpreadsheetApp.getActive();
    var allsheets = ss.getSheets();
    for (var s in allsheets){
      var sheet=allsheets[s]
      var maxColumns = sheet.getMaxColumns(); 
      var lastColumn = sheet.getLastColumn();
      if (maxColumns-lastColumn != 0){
        sheet.deleteColumns(lastColumn+1, maxColumns-lastColumn);
      }
    }
  }
  
  function removeEmptyRowsAndColumns() {
    removeEmptyRows();
    removeEmptyColumns();
  }
